model User {
  id        Int      @id @default(autoincrement())
  name      String
  email     String   @unique
  password  String
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  role  Role    @default(ADMIN)
  aktif Boolean @default(true)

  @@map("user")
}
